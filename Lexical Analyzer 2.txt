#include <iostream>
using namespace std;

int main() {

    char op[] = {'+', '-', '*', '/', '%', '='};

    string s;

    cout<<"Enter input: ";
    cin>>s;
    int count =0;

    for(int i =0;s[i]!='\0';i++){
            for(int j=0;j<6;j++){
               if(s[i]== op[j]){
                    count++;
                cout<<"Operator"<<count<<" : "<<op[j]<<endl;
               }
            }
    }
    if(count==0){
        cout<<"No operators";
    }

    return 0;
}
